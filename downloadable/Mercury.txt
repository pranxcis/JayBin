import bpy
import os
import math

# ==== SET TEXTURE PATH ====
mercury_texture_path = r"C:\Users\Francis Balbin\Downloads\Francis\Solar System Blender\Planets Resources\Mercury.jpg"  # <== Change this

# ==== CLEAN SCENE ====
bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)
bpy.ops.outliner.orphans_purge(do_local_ids=True, do_linked_ids=True, do_recursive=True)

# ==== CREATE MERCURY SPHERE ====
bpy.ops.mesh.primitive_uv_sphere_add(segments=128, ring_count=64, radius=1, location=(0, 0, 0))
planet = bpy.context.active_object
planet.name = "Mercury"
bpy.ops.object.shade_smooth()

# ==== CREATE MATERIAL ====
material = bpy.data.materials.new(name="MercuryMaterial")
material.use_nodes = True
nodes = material.node_tree.nodes
links = material.node_tree.links

# Clear default nodes
for node in nodes:
    nodes.remove(node)

# Create nodes
tex_image = nodes.new(type='ShaderNodeTexImage')
tex_image.image = bpy.data.images.load(mercury_texture_path)
tex_image.location = (-500, 0)

bsdf = nodes.new(type='ShaderNodeBsdfPrincipled')
bsdf.location = (-200, 0)

output = nodes.new(type='ShaderNodeOutputMaterial')
output.location = (200, 0)

# Link nodes
links.new(tex_image.outputs['Color'], bsdf.inputs['Base Color'])
links.new(bsdf.outputs['BSDF'], output.inputs['Surface'])

# Assign material to planet
planet.data.materials.append(material)

# ==== ADD SUNLIGHT ====
bpy.ops.object.light_add(type='SUN', location=(5, -5, 5))
sun = bpy.context.active_object
sun.data.energy = 8
sun.rotation_euler = (0.9, 0, 0.8)

# ==== ADD CAMERA ====
bpy.ops.object.camera_add(location=(0, -6, 0), rotation=(1.5708, 0, 0))  # Looking directly at the center
camera = bpy.context.active_object
bpy.context.scene.camera = camera

# Adjust camera lens (make view wider)
camera.data.lens = 35  # You can go lower (e.g., 28) for a wider view

# ==== RENDER SETTINGS ====
bpy.context.scene.render.engine = 'CYCLES'
bpy.context.scene.cycles.device = 'GPU'  # or 'CPU'
bpy.context.scene.render.film_transparent = False

# ==== ANIMATION (10 seconds, proportional rotation) ====
scene = bpy.context.scene
scene.frame_start = 1
scene.frame_end = 240
scene.render.fps = 24

planet.rotation_mode = 'XYZ'

# Start rotation
planet.rotation_euler[2] = 0
planet.keyframe_insert(data_path="rotation_euler", frame=1)

# End rotation (360 degrees)
planet.rotation_euler[2] = math.radians(360)
planet.keyframe_insert(data_path="rotation_euler", frame=240)

# Make rotation linear (constant speed)
for fcurve in planet.animation_data.action.fcurves:
    for keyframe in fcurve.keyframe_points:
        keyframe.interpolation = 'LINEAR'

print("✅ Mercury planet created with full view in camera and rotation animation.")
