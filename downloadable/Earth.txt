import bpy

# ==== TEXTURE PATH ====
earth_texture_path = r"C:\Users\Francis Balbin\Downloads\Francis\Solar System Blender\Planets Resources\Earth.jpg" # ✅ Replace with your actual path

# ==== CLEAN SCENE ====
bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)
bpy.ops.outliner.orphans_purge(do_local_ids=True, do_linked_ids=True, do_recursive=True)

# ==== CREATE EARTH SPHERE ====
bpy.ops.mesh.primitive_uv_sphere_add(segments=128, ring_count=64, radius=1, location=(0, 0, 0))
earth = bpy.context.active_object
earth.name = "Earth"
bpy.ops.object.shade_smooth()

# ==== CREATE MATERIAL ====
material = bpy.data.materials.new(name="EarthMaterial")
material.use_nodes = True
nodes = material.node_tree.nodes
links = material.node_tree.links

# Clear default nodes
nodes.clear()

# Add nodes
tex_image = nodes.new(type='ShaderNodeTexImage')
tex_image.image = bpy.data.images.load(earth_texture_path)
tex_image.location = (-500, 0)

bsdf = nodes.new(type='ShaderNodeBsdfPrincipled')
bsdf.location = (-200, 0)

output = nodes.new(type='ShaderNodeOutputMaterial')
output.location = (200, 0)

# Connect nodes
links.new(tex_image.outputs['Color'], bsdf.inputs['Base Color'])
links.new(bsdf.outputs['BSDF'], output.inputs['Surface'])

# Assign material
earth.data.materials.append(material)

# ==== LIGHT ====
bpy.ops.object.light_add(type='SUN', location=(5, -5, 5))
sun = bpy.context.active_object
sun.data.energy = 8
sun.rotation_euler = (0.9, 0, 0.8)

# ==== CAMERA ====
bpy.ops.object.camera_add(location=(0, -6, 0), rotation=(1.5708, 0, 0))
camera = bpy.context.active_object
camera.data.lens = 35
bpy.context.scene.camera = camera

# ==== RENDER SETTINGS ====
bpy.context.scene.render.engine = 'CYCLES'
bpy.context.scene.cycles.device = 'GPU'
bpy.context.scene.render.film_transparent = False

print("✅ Earth created with surface texture and proper lighting.")

# ==== ANIMATION SETTINGS ====
scene = bpy.context.scene
scene.frame_start = 1
scene.frame_end = 120  # 5 seconds at 24 fps
scene.render.fps = 24

# ==== EARTH ROTATION (REALISTIC DIRECTION) ====
# Earth rotates once every ~24 hours, but we scale it for visibility

earth.rotation_euler[2] = 0
earth.keyframe_insert(data_path="rotation_euler", frame=1)

earth.rotation_euler[2] = 6.28319  # 360° in radians
earth.keyframe_insert(data_path="rotation_euler", frame=120)

# ==== LINEAR INTERPOLATION (CONSTANT SPEED) ====
action = earth.animation_data.action
for fcurve in action.fcurves: 
    for keyframe in fcurve.keyframe_points:
        keyframe.interpolation = 'LINEAR'

# ==== OUTPUT SETTINGS (PNG SEQUENCE) ====
scene.render.film_transparent = True

# ==== RENDER OPTIMIZATION ====
scene.render.use_persistent_data = True  # faster animation rendering

# Cycles quality/performance balance
scene.cycles.samples = 128
scene.cycles.preview_samples = 32
scene.cycles.use_adaptive_sampling = True

# Denoising (important for animation)
scene.cycles.use_denoising = True
scene.cycles.denoiser = 'OPENIMAGEDENOISE'

print("✅ PNG sequence + render optimization applied.")
