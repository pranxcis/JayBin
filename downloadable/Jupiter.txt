import bpy

# ==== TEXTURE PATH ====
jupiter_texture_path = r"C:\Users\Francis Balbin\Downloads\Francis\Solar System Blender\Planets Resources\Jupiter.jpg"  # 🔁 Replace with your path

# ==== CLEAN SCENE ====
bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)
bpy.ops.outliner.orphans_purge(do_local_ids=True, do_linked_ids=True, do_recursive=True)

# ==== CREATE JUPITER SPHERE ====
bpy.ops.mesh.primitive_uv_sphere_add(segments=128, ring_count=64, radius=1.0, location=(0, 0, 0))
jupiter = bpy.context.active_object
jupiter.name = "Jupiter"
bpy.ops.object.shade_smooth()

# ==== MATERIAL SETUP ====
mat = bpy.data.materials.new(name="JupiterMaterial")
mat.use_nodes = True
nodes = mat.node_tree.nodes
links = mat.node_tree.links
nodes.clear()

# Add nodes
tex_image = nodes.new(type='ShaderNodeTexImage')
tex_image.image = bpy.data.images.load(jupiter_texture_path)
tex_image.location = (-500, 0)

bsdf = nodes.new(type='ShaderNodeBsdfPrincipled')
bsdf.location = (-200, 0)

output = nodes.new(type='ShaderNodeOutputMaterial')
output.location = (200, 0)

# Connect nodes
links.new(tex_image.outputs['Color'], bsdf.inputs['Base Color'])
links.new(bsdf.outputs['BSDF'], output.inputs['Surface'])

# Assign material
jupiter.data.materials.append(mat)

# ==== LIGHT ====
bpy.ops.object.light_add(type='SUN', location=(5, -5, 5))
sun = bpy.context.active_object
sun.data.energy = 2
sun.rotation_euler = (0.9, 0, 0.8)
sun.data.angle = 0.02  # optional softer shadows

# ==== CAMERA ====
bpy.ops.object.camera_add(location=(0, -6, 0), rotation=(1.5708, 0, 0))
camera = bpy.context.active_object
camera.data.lens = 35
bpy.context.scene.camera = camera

# ==== RENDER SETTINGS ====
bpy.context.scene.render.engine = 'CYCLES'
bpy.context.scene.cycles.device = 'GPU'
bpy.context.scene.render.film_transparent = False

print("✅ Jupiter created with surface texture and lighting.")

# ==== ANIMATION SETTINGS ====
scene = bpy.context.scene
scene.frame_start = 1
scene.frame_end = 120  # 5 sec at 24 FPS
scene.render.fps = 24

# ---- JUPITER ROTATION ----
jupiter.rotation_euler[2] = 0
jupiter.keyframe_insert(data_path="rotation_euler", frame=1)

# Jupiter spins faster → give it more rotation
jupiter.rotation_euler[2] = 12.56637  # 720 degrees (2 full spins)
jupiter.keyframe_insert(data_path="rotation_euler", frame=120)

# ---- LINEAR INTERPOLATION ----
action = jupiter.animation_data.action
for fcurve in action.fcurves: 
    for keyframe in fcurve.keyframe_points:
        keyframe.interpolation = 'LINEAR'
        
# ==== RENDER OPTIMIZATION ====
scene.render.use_persistent_data = True

scene.cycles.samples = 128
scene.cycles.preview_samples = 32
scene.cycles.use_adaptive_sampling = True

scene.cycles.use_denoising = True
scene.cycles.denoiser = 'OPENIMAGEDENOISE'

print("✅ Jupiter animation + render settings applied.")
