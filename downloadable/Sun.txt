import bpy
import os
import math

# ==== SET TEXTURE PATH ====
sun_texture_path = r"C:\Users\Francis Balbin\Downloads\Francis\Solar System Blender\Planets Resources\Sun.jpg"  # <== Change this

# ==== CLEAN SCENE ====
bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)
bpy.ops.outliner.orphans_purge(do_local_ids=True, do_linked_ids=True, do_recursive=True)

# ==== CREATE SUN SPHERE ====
bpy.ops.mesh.primitive_uv_sphere_add(segments=128, ring_count=64, radius=1.5, location=(0, 0, 0))
sun_obj = bpy.context.active_object
sun_obj.name = "Sun"
bpy.ops.object.shade_smooth()

# ==== CREATE MATERIAL ====
material = bpy.data.materials.new(name="SunMaterial")
material.use_nodes = True
nodes = material.node_tree.nodes
links = material.node_tree.links

# Clear default nodes
for node in nodes:
    nodes.remove(node)

# Create nodes
tex_image = nodes.new(type='ShaderNodeTexImage')
tex_image.image = bpy.data.images.load(sun_texture_path)
tex_image.location = (-600, 0)

bright = nodes.new(type='ShaderNodeBrightContrast')
bright.location = (-300, 0)
bright.inputs['Bright'].default_value = 3.0
bright.inputs['Contrast'].default_value = 8

hue = nodes.new(type='ShaderNodeHueSaturation')
hue.location = (-450, 0)
hue.inputs['Saturation'].default_value = 1.3

# 🔥 Emission instead of BSDF
emission = nodes.new(type='ShaderNodeEmission')
emission.location = (0, 0)
emission.inputs['Strength'].default_value = 0.1  # adjust if too bright/dim

output = nodes.new(type='ShaderNodeOutputMaterial')
output.location = (200, 0)

# Link nodes
links.new(tex_image.outputs['Color'], bright.inputs['Color'])
links.new(bright.outputs['Color'], emission.inputs['Color'])
links.new(emission.outputs['Emission'], output.inputs['Surface'])

links.new(tex_image.outputs['Color'], hue.inputs['Color'])
links.new(hue.outputs['Color'], bright.inputs['Color'])

# Assign material
sun_obj.data.materials.append(material)

# ==== ADD CAMERA ====
bpy.ops.object.camera_add(location=(0, -7, 0), rotation=(1.5708, 0, 0))
camera = bpy.context.active_object
bpy.context.scene.camera = camera
camera.data.lens = 35

# ==== RENDER SETTINGS ====
scene = bpy.context.scene
scene.render.engine = 'CYCLES'
scene.cycles.device = 'GPU'
scene.render.film_transparent = False

# ==== RENDER OPTIMIZATION ====
scene.render.use_persistent_data = True
scene.cycles.samples = 128
scene.cycles.preview_samples = 32
scene.cycles.use_adaptive_sampling = True

scene.cycles.use_denoising = True
scene.cycles.denoiser = 'OPENIMAGEDENOISE'

# ==== ANIMATION (5 seconds, 24 FPS) ====
scene.frame_start = 1
scene.frame_end = 120
scene.render.fps = 24

sun_obj.rotation_mode = 'XYZ'

# Start rotation
sun_obj.rotation_euler[2] = 0
sun_obj.keyframe_insert(data_path="rotation_euler", frame=1)

# End rotation
sun_obj.rotation_euler[2] = math.radians(360)
sun_obj.keyframe_insert(data_path="rotation_euler", frame=120)

# Linear interpolation
for fcurve in sun_obj.animation_data.action.fcurves:
    for keyframe in fcurve.keyframe_points:
        keyframe.interpolation = 'LINEAR'

print("☀️ Sun created with emission (evenly lit, no shadows).")
